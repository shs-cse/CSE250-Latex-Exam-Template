\titledquestion{CO3}[10]
What is the value of equivalent resistance $R_{eq}$? \textit{[Must show step by step procedure of finding $R_{eq}$]}

\begin{circuitikz}[scale=3]
    \draw
    (1,-1) -- (0,-1)
    to[open, o-o, l=$R_{eq}$] (0,0)
    (0,0)   to[R=30<\kO>] (1,0)
    (1,0)   to[R, l_=40<\kO>] (1,-1)
    (1,0)   to[R=20<\kO>] (2,0)
    (1,-1)   to[R=40<\kO>] (2,0)
    (1,-1)   to[R=20<\kO>] (2,-1)
    (2,-1)   to[R, l=\rotatebox{90}{30\,\si\kO}] (2,0)
    (2,-1)   to[R=40<\kO>] (3,-1)
    (2,0)   to[R=50<\kO>] (3,0)
    (2,0)   to[R=30<\kO>] (3,-1)
    (3,0)   to[R=10<\kO>] (3,-1)
    ;
\end{circuitikz}

\begin{solutionbox}{24pt}
    50\,\si\kO
\end{solutionbox}
